import React from 'react';

/**
 * Mały komponent pokazujący komunikat po wysłaniu formularza:
 * - Netlify może przekierować np. na /contact?success=1
 * - Poniżej wykrywamy `?success=1` i wyświetlamy box z potwierdzeniem
*/
function SuccessNotice() {
if (typeof window === 'undefined') return null;
const params = new URLSearchParams(window.location.search);
const success = params.get('success') === '1';
if (!success) return null;
return (
<div
role="status"
aria-live="polite"
style={{
        margin: '1rem 0 1.25rem',
        padding: '0.875rem 1rem',
        borderRadius: 8,
        border: '1px solid var(--ifm-color-success-contrast-background, #c8e6c9)',
        background: 'var(--ifm-color-success-contrast-background, #e8f5e9)',
        color: 'var(--ifm-color-success-contrast-foreground, #1b5e20)',
        fontWeight: 600,
      }}
>
Dziękujemy! Twoja wiadomość została wysłana ✅
</div>
);
}
